Section C — Code Typing [30 marks]

Q8. Hard balanced brackets with ignored characters. [10 marks]

Write bool isBalanced(string s). It must ignore letters, digits, spaces, and operators, and only check (), {}, [].

bool isBalanced(string s)
{
    stack<char> st;

    for(char ch : s)
    {
        if(ch == '(' || ch == '{' || ch == '[')
            st.push(ch);
        else if(ch == ')' || ch == '}' || ch == ']')
        {
            if(st.empty())
                return false;

            char open = st.top();
            st.pop();

            if((ch == ')' && open != '(') ||
               (ch == '}' && open != '{') ||
               (ch == ']' && open != '['))
                return false;
        }
    }

    return st.empty();
}
Q9. Hard linear probing search with deleted cells. [10 marks]

Write int searchLinearProbing(int table[], int size, int key). Empty = -1. Deleted = -2. Searching must continue through -2 but stop at -1 or full loop.

int searchLinearProbing(int table[], int size, int key)
{
    int index = key % size;
    int start = index;

    while(table[index] != -1)
    {
        if(table[index] != -2 && table[index] == key)
            return index;

        index = (index + 1) % size;

        if(index == start)
            break;
    }

    return -1;
}
Important: Do not stop at -2 because deleted cells may be inside a collision chain.
Q10. Hard deleteTail: return deleted value and handle all cases. [10 marks]

Given Node, write bool deleteTail(Node*& head, int& deletedValue). Return false if list empty. Otherwise delete last node, store its value in deletedValue, and return true.

struct Node{
    int data;
    Node* next;
};
bool deleteTail(Node*& head, int& deletedValue)
{
    if(head == NULL)
        return false;

    if(head->next == NULL)
    {
        deletedValue = head->data;
        delete head;
        head = NULL;
        return true;
    }

    Node* current = head;

    while(current->next->next != NULL)
        current = current->next;

    Node* tail = current->next;
    deletedValue = tail->data;
    delete tail;
    current->next = NULL;

    return true;
}

Q14. Coding question — recursive BST search. [10 marks]

Write a recursive function:

bool search(Node* root, int key)

Requirements:

  • Use recursion only
  • Return true if key exists
  • Use BST property to reduce searching
  • No loops allowed
bool search(Node* root, int key)
{
    if(root == NULL)
        return false;

    if(root->data == key)
        return true;

    if(key < root->data)
        return search(root->left, key);

    return search(root->right, key);
}
